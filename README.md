## Install

```bash
npm install --save auto-responsive-grid
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'auto-responsive-grid'
import 'auto-responsive-grid/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
